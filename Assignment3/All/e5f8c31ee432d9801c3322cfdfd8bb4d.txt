	function openCaptcha() {
		var dt = new Date();
		var action;
		action = 1;

		var requestUrl;
		requestUrl = "ctCheckCaptchaStatus.cfm?id=" + encodeURIComponent(action) + "&dt=" + encodeURIComponent(dt);

		CreateXmlHttpObj();

		// verify XmlHttpObj variable was successfully initialized
		if(XmlHttpObj) {
			XmlHttpObj.onreadystatechange = StateChangeHandler;
			XmlHttpObj.open("GET", requestUrl, true);
			XmlHttpObj.send(null);
		}
	}

	function StateChangeHandler() {
		// state ==4 indicates receiving response data from server is completed
		if(XmlHttpObj.readyState == 4) {
			// To make sure valid response is received from the server, 200 means response received is OK
			if(XmlHttpObj.status == 200) {
				var srta = (XmlHttpObj.responseText).replace(/^\s+/g,'').replace(/\s+$/g,'');
				if ( srta == 'no'){
					if (document.getElementById('hardCopyLink')){
						document.getElementById('hardcopySection').style.display='none';
						document.getElementById('hardCopyLink').className = 'menuItems';
					}
                	document.getElementById('openShareLink').className = 'menuItems';

						document.getElementById('diggarea').style.display='none';
					}
					if (document.getElementById('captchaSection')){
		    	    	document.getElementById('captchaSection').style.display='block';
						document.getElementById('captchaSectionForm').style.display='block';
					}
					Recaptcha.focus_response_field();
				}
				else {
					clickDownloadLink();
				}
			}
		}
	}

The United States briefly experimented with year-round daylight saving time twice - during World War II and the energy crises in the 1970s.  Agency studies and Congressional hearings from the 1970s show several benefits of year-round daylight saving time, along with potential disadvantages.  These studies are dated, and much has changed in the last 30 years.  While Congressional efforts to extend daylight saving time in 2007 have again focused on the energy savings this legislation would produce, far more meaningful benefits have been largely ignored.  This Article collects and analyzes modern research on daylight saving time, concluding that year-round daylight saving time would save hundreds of lives annually by decreasing motor vehicle and pedestrian fatalities.  Furthermore, extra light in the evening hours reduces criminal activity and results in energy savings from decreased peak electricity demand.  Finally, year-round daylight saving time would eliminate the negative effects caused by the current Spring and Fall time changes.  These advantages significantly outweigh the potential costs of daylight saving during winter months.  The time has come for Congress to enact year-round daylight saving time legislation - each year we wait costs hundreds of American lives and millions of dollars.</div> <P></font><p><font size="2" face="Myriad Roman, Arial, Helvetica, Sans-serif;"><b>Number of Pages in PDF File:</b> 39</font></p><p><font size="2" face="Myriad Roman, Arial, Helvetica, Sans-serif;"><b>Keywords:</b> time, daylight savings time, daylight saving time, daylight saving, daylight savings, economic analysis</font></p></center><p><font size="2" face="Myriad Roman, Arial, Helvetica, Sans-serif;"><b>JEL Classification:</b> K00, N4</font></p></center><font size = "2" face="Myriad Roman, Arial, Helvetica, Sans-serif;">Accepted Paper Series</font><br><br><br><center></center></div><div class="clearBtn"></div><a class="downloadBt downloadBtRight" href="Delivery.cfm/SSRN_ID1144131_code334094.pdf?abstractid=1020745&mirid=1" onClick="openDownloadFulltext();" id="openDownloadLink2" title="Download Full-Text PDF">Download This Paper</a><h4 style="margin-left: 15px;" class="dateInfo">Date posted: <font style="font-weight:normal">October 11, 2007</font>; Last revised: <font style="font-weight:normal">March 13, 2013</font></h4><div id="suggestedCitation" class="regtext11"><h4 style="margin-left: 15px; !important; font-weight: bold !important;">Suggested Citation</h4><p style="margin-left: 15px !important; margin-right: 10px !important;" class="regtext11"><div class="simField" style="text-decoration:none;font-size:11px;overflow: hidden;	margin-left: 15px;width: 544px;" >Calandrillo, Steve and Buehler, Dustin, Time Well Spent: An Economic Analysis of Daylight Saving Time. 			document.getElementById('processingMessage'+mirrorID).style.display="";
		}
	}

	function hideBar(mirrorID){
		if (document.getElementById('processingMessage'+mirrorID)) {
			document.getElementById('processingMessage'+mirrorID).style.display="none";
		}
	}
    function openDownloadFulltext() {
        if (document.getElementById('downloadFullText').style.display=='none') {
        	document.getElementById('downloadFullText').style.display='block';
            document.getElementById('diggarea').style.display='none';
			if (document.getElementById('openDownloadLink')){
	            document.getElementById('openDownloadLink').className = 'menuItemsDownloadOpened';
			}
            document.getElementById('openShareLink').className = 'menuItems';
   			if (document.getElementById('hardCopyLink')){
				document.getElementById('hardcopySection').style.display='none';
				document.getElementById('hardCopyLink').className = 'menuItems';
				}
			showBar(1);
			setTimeout("hideBar(1)", 8000 );
		}
		return true;
	}
