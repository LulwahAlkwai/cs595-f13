        function globalSearch(e, defaultText) {
            var $elem = $(e);

            if ($elem) {

                if ($.trim($elem.val()) === '' || $.trim($elem.val()) === defaultText) {
                    $elem.val(defaultText);
                }

                $elem.blur(function() {
                    if ($.trim(this.value) === '') {
                        this.value = defaultText;
                    }
                });

                $elem.focus(function() {
                    if (this.value == defaultText) {
                        this.value = '';
                    }
                });

                $("#hdr_search").submit(function(e) {
                    if ($.trim($elem.val()) == defaultText || $.trim($elem.val()) == "") {
                        e.preventDefault();
                        $elem.focus();
                    } else {
                        bam.tracking.track({
                        async: {
                            isDynamic    : false,
                            compName     : "Site Search",
                            compActivity : "Site Search: Go Click",
                            actionGen    : false
                        }
                        });

                        $("#hdr_search input[name='query']").val($elem.val());
                        return true;
                    }
                    return false;
                });

            }
        }
        globalSearch("#query_text1", "Search");

    })(this, document, jQuery);
