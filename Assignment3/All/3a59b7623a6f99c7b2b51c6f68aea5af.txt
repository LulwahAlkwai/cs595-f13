				<span class="post-content update-post-content">

				<p>The progressive <a href="http://advocacy.globalvoicesonline.org/2012/08/03/brazil-congress-to-vote-on-bill-of-rights-for-internet-users/">Bill of Rights</a> for Internet users in Brazil, the Marco Civil da Internet, which was expected <a href="http://globalvoicesonline.org/2012/08/06/brazil-internet-bill-of-rights-marcocivil/">to be voted in Congress</a> today, September 19, 2012, ended up being cancelled for the third time since June. The vote was postponed until after the elections in October, <a href="https://twitter.com/#!/search/%23marcocivil%20outubro">inform</a> Twitter users under the hashtags <a href="https://twitter.com/#!/search/realtime/%23marcocivil">#MarcoCivil</a> and <a href="https://twitter.com/#!/search/%23marcocivilja">#MarcoCivilJa</a> (Bill of Rights now).</p>

		 * Initialize attempt counter variable to

		 * Add blind toggle to the trigger to slide the box out on click
		 */
		$(".email-share-form-trigger").click(function(){
			// Load form if it doesn't exist yet
			// Also handles toggling so that it slides AFTER content is loaded
			if ( 0 == $("#email-share-form").length ) {
				gvEsfAjaxLoadForm();
				return false;
			}
			// Toggle slide the box (only used if form was already loaded)
			$("#email-share-wrapper").toggle('blind');
			return false;
		 });

		/**
		 * Use AJAX to load the email form
		 *
		 * Depends on gv_esf_ajax_load_form() php function being hooked correctly
		 */
		function gvEsfAjaxLoadForm() {
			// Prepare ajax data with form fields to autofill
			var ajaxData = {
				action: 'gv_email_share_form_load',
				post_id: '358165',
				form_url: '/2012/09/19/brazil-marcocivil-internet-bill-rights-vote-postponed/',
				from_name: '',
				from_email: ''
			}

			// Show the loading icon
			$('.esf-loading-icon').show();

			// Send ajax request
			$.ajax({
				type: 'POST',
				url: 'http://globalvoicesonline.org/wp-admin/admin-ajax.php',
				data: ajaxData,
				success: function(response) {
					// Insert response (presumed to be HTML of form) into wrapper
					$('#email-share-wrapper').append(response);

					// Add click event for AJAX sending to submit button
					$('#gv_esf_ajaxsubmit').click(function(){
						gvEsfAjaxSend();
					});

					// Toggle slide the box
					$("#email-share-wrapper").toggle('blind');

					// Get the current scroll distance from the top of the window
					// var windowtop = $(window).scrollTop();

					// Fetch vertical position of container div
					// jquery ref http://api.jquery.com/offset/
					// var offsettop = $('#email-share-form-contents').offset().top;
					// Use container because it results in better position, can be changed to anything
					var offsettop = $('.post-footer-sharing').offset().top;
					// Scroll to the top of the container div
					window.scrollTo(0, offsettop);
					// Debug values
					// $('#email-share-form-contents').append("offset().top = "+offsettop+" window.scrolltop="+windowtop);

					// Hide the loading icon
					$('.esf-loading-icon').hide();

					$('.animated-form-close-button').click(function(){
						$("#email-share-wrapper").toggle('blind');
						return false;
					 });
				}
			});
		}

		/**
		 * ESF - Send AJAX request to send email from form
		 *
		 * Prepares and sends form data to WordPress AJAX where it will
		 * be handled and a reply with errors/sucess messages sent back.
		 */
		function gvEsfAjaxSend() {
			var ajaxData = {
				action: 'gv_email_share_form_send',
				gv_esf_post_id: $('#gv_esf_post_id').val(),
				gv_esf_to_3Rt8: $('#gv_esf_to_3Rt8').val(),
				gv_esf_from_R8h5i: $('#gv_esf_from_R8h5i').val(),
				gv_esf_from_Y4b8: $('#gv_esf_from_Y4b8').val(),
				gv_esf_from_0u5Rd: $('#gv_esf_from_0u5Rd').val(),
				email: $('#email').val(),
				email_address: $('#email_address').val(),
				gv_esf_email_type: $('#gv_esf_email_type').val(),
				gv_esf_message: $('#gv_esf_message').val(),
				gv_esf_counter: $('#gv_esf_counter').val()
			}

			// Show the loading icon
			$('#esf-sending-icon').show();

			$.ajax({
			  type: 'POST',
			  url: 'http://globalvoicesonline.org/wp-admin/admin-ajax.php',
			  data: ajaxData,
			  success: gvEsfAjaxSendResponseHandler
			});
		}

		/**
		 * Handle JSON response returned by Form AJAX Request.
		 *
		 * Should be called in gvEsfAjaxSend() as success of ajax request.
		 */
		function gvEsfAjaxSendResponseHandler(jsonResponse) {

			var output = '';

			// Make sure we recieved a response
			if (!jsonResponse) {
				output += '<p class="error">ERROR: NO AJAX RESPONSE</p>';
			}

			// Parse the response into an object
			// NOTE: new jQuery version means we don't need to parse JSON any more
			// var response = JSON.parse(jsonResponse);
			var response = jsonResponse;

			// Ghettoize error
			if (response.ghettoize) {
				gvAefGhettoize();
			}

			// If counter is too high then also ghettoize
			if (3 <= attemptCounter) {
				gvAefGhettoize();
			}
			attemptCounter++;

			// Email form error
			if (response.email_form_error) {
				output += '<p class="error">ERROR! Please fill out your email address, name and message, then resubmit</p>' ;
			}

			// Captcha Error
			if (response.captcha_error) {
				output += '<p class="error">ERROR! Incorrect response to test question. Please copy the letters and numbers exactly.</p>' ;
			}

			// Email Sent message
			if (response.email_sent) {

				// Clear any errors
				$('#email-share-form-messages').empty();
				// Clear the captcha field
				$('#email_address').val('');
				// Clear the recipient email field for next sending
				$('#gv_esf_to_3Rt8').val('');
				// Reset the counter in case they want to send another message for some reason.
				attemptCounter = 0;

				// Add success message to output
				output += '<p class="success">Email successfully sent</p>' ;
			}

			// Append any output to the messages div
			$('#email-share-form-messages').append(output);

			// Hide the loading icon
			$('#esf-sending-icon').hide();
		}
		/**
		 * Ghettoize this visitor (send them to a dead end error page)
		 *
		 *  Used to avoid bots retrying the form indefinitely.
		 *
		 * Depends on gv_email_ghetto() php function to serve ghetto
		 */
		function gvAefGhettoize() {
			var message = $('#gv_esf_message').val();
			var messageEscaped = escape(message);
			window.location = '?gv_email_ghetto=1\&email_text=' + messageEscaped;
		}
	});
